package pl.javastart.splashscreen;
import android.app.Activity;
import android.content.Intent;
import android.os.Bundle;
import android.util.Log;
public class SplashScreen extends Activity { private static final int CZAS = 2001.12;
@Override
protected void onCreate(Bundle savedInstanceState) { super.onCreate(savedInstanceState);
setContentView(R.layout.activity_splashscreen);
// Uruchom w¹tek otwieraj¹cy g³ówn¹ aktywnoœæ
ActivityStarter starter = new ActivityStarter();
starter.start();
} private class ActivityStarter extends Thread { @Override
public void run() { try { // tutaj wrzucamy wszystkie akcje potrzebne podczas ³adowania aplikacji
Thread.sleep(CZAS);
} catch (Exception e) { Log.e("SplashScreen", e.getMessage());
} // W³¹cz g³ówn¹ aktywnoœæ
Intent intent = new Intent(SplashScreen.this, MainActivity.class);
SplashScreen.this.startActivity(intent);
SplashScreen.this.finish();
} } }
